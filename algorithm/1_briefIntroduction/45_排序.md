package main

import "fmt"

func main() {
	fmt.Println(sort1([]int{2, 1, 3, 5, 4, 7, 8, 6}))
	fmt.Println(sort2([]int{2, 1, 3, 5, 4, 7, 8, 6}))
	fmt.Println(sort3([]int{2, 1, 3, 5, 4, 7, 8, 6}))
}

// 冒泡排序 o(n^2)
func sort1(nums []int) []int {
	l := len(nums)

	for i := 0; i < l-1; i++ {
		for j := i + 1; j < l; j++ {
			if nums[i] > nums[j] {
				nums[i] ^= nums[j]
				nums[j] ^= nums[i]
				nums[i] ^= nums[j]
			}
		}
	}
	return nums
}

// 插入排序 o(n^2)
func sort2(nums []int) []int {
	l := len(nums)
	for i := 1; i < l; i++ {
		tmpIndex := i - 1
		for tmpIndex >= 0 && nums[tmpIndex] > nums[tmpIndex+1] {
			nums[tmpIndex] ^= nums[tmpIndex+1]
			nums[tmpIndex+1] ^= nums[tmpIndex]
			nums[tmpIndex] ^= nums[tmpIndex+1]
			tmpIndex--
		}
	}
	return nums
}

// 选择排序 o(n^2)

// 快速排序
func sort3(nums []int) []int {
	sort(nums,0,len(nums)-1)
	return nums
}

func sort(nums []int, l int, r int) {
	if l > r {
		return
	}
	tmp := nums[l]
	i := l
	j := r
	for i != j {
		for i < j && tmp <= nums[j] {
			j--
		}

		for i < j && tmp >= nums[i] {
			i++
		}

		t := nums[i]
		nums[i] = nums[j]
		nums[j] = t
	}

	nums[l] = nums[i]
	nums[i] = tmp

	sort(nums, l, i-1)
	sort(nums, i+1, r)
}
