- 非递归算法
```
二分查找时，每次都在原有查找内容进行二分，所以时间复杂度为O（log2 n）
因为变量值创建一次，所以空间复杂度为O（1）
```

- 递归算法

```
时间复杂度为O（log2 n）
每进行一次递归都会创建变量，所以空间复杂度为O（log2 n）
```

```
package main

import (
	"fmt"
	"time"
)

func find(array []int, aim int) bool{
	lArray := len(array)
	left := 0
	right := lArray - 1

	mid := 0
	for true {
		if left <= right {
			mid =  left + (right - left)/2
			if array[mid] > aim {
				right = mid - 1
			}else if array[mid] < aim {
				left = mid + 1
			}else{
				return true
			}
		} else {
			return false
		}
	}
	return false
}

func findRecursion(array []int,left int, right int  ,aim int) bool {
	if left <= right {
		mid := left + ( right - left)/2
		if array[mid] > aim {
			return findRecursion(array, left, mid-1, aim)
		}else if array[mid] < aim{
			return findRecursion(array, mid+1, right, aim)
		} else {
			return true
		}
	}else {
		return false
	}
}


func main() {
	array := make([]int, 0)

	for i:=0 ;i < 60000000;i++ {
		array = append(array,i)
	}

	t1 := time.Now()
	aim := 30000000
	if find(array, aim) {
		fmt.Println("exist")
	}else{
		fmt.Println("not exist")
	}

	t2 := time.Now()
	fmt.Println("非递归花费时间:", t2.Sub(t1))

	t3 := time.Now()
	if findRecursion(array,0,len(array)-1, aim) {
		fmt.Println("exist")
	}else{
		fmt.Println("not exist")
	}
	//time.Sleep(time.Second)
	t4 := time.Now()
	fmt.Println("递归花费时间:", t4.Sub(t3))



}
```
